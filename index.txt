  <script>
    const KEYWORDS = new Set([
      "and","break","do","else","elseif","end","false","for","function",
      "if","in","local","nil","not","or","repeat","return","then","true",
      "until","while","self","print","tostring","tonumber","type","pairs",
      "ipairs","next","error","assert","pcall","xpcall","select","unpack",
      "table","string","math","os","io","coroutine","rawget","rawset",
      "rawequal","rawlen","require","load","loadstring","dofile","loadfile",
      "collectgarbage","gcinfo","newproxy","setfenv","getfenv","setmetatable",
      "getmetatable","typeof","task","game","workspace","script"
    ]);

    function obfuscateLua(code, checks) {
      let result = code;
      const strings = [];

      result = result.replace(/"((?:[^"\\]|\\.)*)"|'((?:[^'\\]|\\.)*)'/g, (_, d, s) => {
        const str = d !== undefined ? d : s;
        const encoded = str.split("").map(c => "\\" + c.charCodeAt(0)).join("");
        const idx = strings.length;
        strings.push('"' + encoded + '"');
        return "__S" + idx + "__";
      });

      const varMap = {};
      let varCounter = 0;

      result = result.replace(/\blocal\s+(\w+)/g, (match, name) => {
        if (!KEYWORDS.has(name) && !varMap[name]) {
          varCounter++;
          varMap[name] = "_v" + varCounter.toString(36);
        }
        return "local " + (varMap[name] || name);
      });

      for (const [orig, renamed] of Object.entries(varMap)) {
        result = result.replace(new RegExp("\\b" + orig + "\\b", "g"), renamed);
      }

      result = result.replace(/__S(\d+)__/g, (_, i) => strings[parseInt(i)]);

      if (checks) {
        result = "local _chk=coroutine.wrap(function()while true do coroutine.yield()end end)\n" + result;
      }

      return "-- Obfuscated by kers0ne obfuscator\n" + result;
    }

    function runObfuscate() {
      const code = document.getElementById("input").value.trim();
      if (!code) return;

      const btn = document.getElementById("obfuscateBtn");
      btn.disabled = true;
      btn.textContent = "Obfuscating...";

      setTimeout(() => {
        try {
          const checks = document.getElementById("checkToggle").checked;
          const obfuscated = obfuscateLua(code, checks);
          document.getElementById("output").value = obfuscated;
        } catch (e) {
          document.getElementById("output").value = "ERROR: " + e.message;
        }

        document.getElementById("outputSection").style.display = "flex";
        btn.style.display = "none";
      }, 10);
    }

    function copyOutput() {
      const el = document.getElementById("output");
      el.select();
      el.setSelectionRange(0, el.value.length);
      try {
        if (navigator.clipboard && window.isSecureContext) {
          navigator.clipboard.writeText(el.value);
        } else {
          document.execCommand("copy");
        }
      } catch {
        document.execCommand("copy");
      }
      const btn = document.getElementById("copyBtn");
      btn.textContent = "Copied!";
      setTimeout(() => { btn.textContent = "Copy"; }, 1500);
    }

    function downloadOutput() {
      const text = document.getElementById("output").value;
      const blob = new Blob([text], { type: "text/plain" });
      const a = document.createElement("a");
      a.href = URL.createObjectURL(blob);
      a.download = "obfuscated.lua";
      a.click();
    }

    function resetAll() {
      document.getElementById("input").value = "";
      document.getElementById("output").value = "";
      document.getElementById("outputSection").style.display = "none";
      const btn = document.getElementById("obfuscateBtn");
      btn.style.display = "block";
      btn.disabled = false;
      btn.textContent = "Obfuscate";
    }
